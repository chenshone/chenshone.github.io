    排序(二) 归并、快速

    CHENSHONE 2020/10/25 算法思维sort

    不作任何说明的话，排序都按升序来。

    # 归并排序 稳定

    实质是把比较行为变成了有序信息并传递，比O(N^2)的排序快。

    适用于要求对于某个数，左边有多少个数比其大/小，右边有多少个数比其大/小，求总和的问题

    # 递归

    # 思路

    归并排序的意思是，将一个大数组分成左右两部分，先将左右两部分分别用递归的方式排好序，然后将它们merge起来，就完成了排序操作。

    merge思路：额外开辟一个与原数组一样大的空间，准备两个指针left和right，分别指向已经排好序的左右两个子数组，依次比较，较小的放入新开的数组，然后其指针往后移一位，直到有一个指针到头，将剩下没有比较的依次放入新数组。最后用新数组覆盖原数组。

    # 代码

    public class MergeSort {
    //    递归
    public static void mergeSort1(int[] arr) {
        if (arr == null || arr.length < 2) {
            return;
        }
        process(arr, 0, arr.length - 1);
    }

    //    将arr的[L,R]范围上，变成有序
    private static void process(int[] arr, int L, int R) {
        if (L == R) {
            return;
        }
        int mid = L + ((R - L) >> 1);
        process(arr, L, mid);
        process(arr, mid + 1, R);
        merge(arr, L, mid, R);
    }

    private static void merge(int[] arr, int l, int mid, int r) {
        int[] helper = new int[r - l + 1];
        int i = 0;
        int p1 = l;
        int p2 = mid + 1;
        while (p1 <= mid && p2 <= r) {
            helper[i++] = arr[p1] <= arr[p2] ? arr[p1++] : arr[p2++];
        }


         // 这两个while只会走一个
        while (p1 <= mid) {
            helper[i++] = arr[p1++];
        }
        while (p2 <= r) {
            helper[i++] = arr[p2++];
        }

        for (i = 0; i < helper.length; i++) {
            arr[l + i] = helper[i];
        }
    }
}

    # 非递归

    # 代码

    public static void mergeSort2(int[] arr) {
    if (arr == null || arr.length < 2) {
        return;
    }
    int n = arr.length;
    int mergeSize = 1; // 一组是2倍的mergeSize，即左数组和右数组的规模
    while (mergeSize < n) {
        int l = 0;
        while (l < n) {
            int mid = l + mergeSize - 1;
            if (mid >= n) { // 左部分都凑不齐，即mid到n的数组已经有序
                break;
            }
            int r = Math.min(mid + mergeSize, n - 1); // 可能出现右数组比左数组小的情况
            merge(arr, l, mid, r);
            l = r + 1;
        }
        if (mergeSize > n / 2) { // 如果n是很大的数，防止mergeSize越界
            break;
        }
        mergeSize <<= 1;
    }
}

    # 快速排序 不稳定

    # 引子：荷兰国旗问题

    荷兰国旗问题就是给定一个数组，给一个划分值n，将数组中小于n的数放左边，等于n的数放中间，大于n的数放右边

    # 代码

    //    以arr[r]做划分值
public static int[] netherlandsFlag(int[] arr, int l, int r) {
    if (l > r) {
        return new int[]{-1, -1};
    }
    if (l == r) {
        return new int[]{l, r};
    }

    int less = l - 1; // 小于区右边界
    int more = r; // 大于区左边界
    int index = l;
    while (index < more) {
        if (arr[index] == arr[r]) {
            index++;
        } else if (arr[index] < arr[r]) {
            swap(arr, index++, ++less); // arr[i]与小于区右一个交换
        } else {
            swap(arr, index, --more); // arr[i]与大于区左一个交换
        }
    }
    swap(arr, more, r); // 将最后的arr[r]，也就是划分值与大于区的第一个交换
    return new int[]{less + 1, more};
}

private static void swap(int[] arr, int a, int b) {
    int tmp = arr[a];
    arr[a] = arr[b];
    arr[b] = tmp;
}

    # 1.0

    # 思路

    对于划分值x(选取区域数组的最后一个)，将其左边的所有数按照小于等于和大于来划分，最后将划分值x与大于区域的第一个交换位置，即可保证x左边为小于等于，右边为大于。

    然后将x左边的区域和右边的区域递归即可完成排序

    # 代码

    public static void quickSort1(int[] arr) {
    if (arr == null || arr.length < 2) {
        return;
    }
    process1(arr, 0, arr.length - 1);
}

private static void process1(int[] arr, int l, int r) {
    if (l > r) {
        return;
    }
    // [l, r] 执行partition
    int mid = partition(arr, l, r);
    process1(arr, l, mid - 1);
    process1(arr, mid + 1, r);
}

private static int partition(int[] arr, int l, int r) {
    if (l > r) {
        return -1;
    }
    if (l == r) {
        return l;
    }

    int lessEqual = l - 1; // 小于等于区右边界
    int index = l;
    while (index < r) {
        if (arr[index] <= arr[r]) {
            swap(arr, index, ++lessEqual);
        }
        index++;
    }
    swap(arr, ++lessEqual, r);
    return lessEqual;
}

    # 2.0

    # 思路

    根据上面的荷兰国旗问题，设计的思路如下：

    1.0思路是每次都是确定一个数，然后将其左右部分递归，2.0的思路是跟荷兰国旗问题一样，将数组划分为小于、等于和大于三个部分，这样，每次递归就只要递归小于和大于两部分即可。

    # 代码

    public static void quickSort2(int[] arr) {
    if (arr == null || arr.length < 2) {
        return;
    }
    process2(arr, 0, arr.length - 1);
}
private static void process2(int[] arr, int l, int r) {
    if (l > r) {
        return;
    }

    int[] equalArea = netherlandsFlag(arr, l, r);
    process2(arr, l, equalArea[0] - 1);
    process2(arr, equalArea[1] + 1, r);
}

    # 3.0

    # 思路

    对于2.0，是默认选中区域的最后边一个数为划分值，3.0是在区域范围内随机选择一个数当做划分值，将其与r位置的数swap，后续步骤与2.0相同。

    # 代码

    public static void quickSort3(int[] arr){
    if (arr == null || arr.length < 2) {
        return;
    }
    process3(arr, 0, arr.length - 1);
}

private static void process3(int[] arr, int l, int r) {
    if (l > r) {
        return;
    }
    swap(arr, l + (int)(Math.random() * (r - l + 1)), r);
    int[] equalArea = netherlandsFlag(arr, l, r);
    process3(arr, l, equalArea[0] - 1);
    process3(arr, equalArea[1] + 1, r);
}
