    # 从左往右的尝试模型

    # 类型1

    规定1和A对应、2和B对应、3和C对应... 那么一个数字字符串比如"111"就可以转化为："AAA"、"KA"、"AK" 给定一个只有数字字符组成的字符串str，返回有多少种转化结果

    # 暴力递归

    public static int number(String str) {
    if (str == null || str.length() == 0) {
        return 0;
    }
    return process(str.toCharArray(), 0);
}

// 从i位置开始往后的一共有多少种结果
private static int process(char[] str, int i) {
    if (i == str.length) { // base case
        return 1;
    }

    // 碰到0 之后的都没法转化
    if (str[i] == '0') {
        return 0;
    }

    if (str[i] == '1') {
        int res = process(str, i + 1); // i自己作为单独一部分转化，后续有多少种解法
        if (i + 1 < str.length) {
            res += process(str, i + 2); // i和i+1两个数作为一个单独的部分，后续有多少种解法
        }
        return res;
    }

    if (str[i] == '2') {
        int res = process(str, i + 1); // i自己作为单独一部分转化，后续有多少种解法

        // 第二个数字超过6就不能转化了
        if (i + 1 < str.length && (str[i + 1] >= '0' && str[i + 1] <= '6')) {
            res += process(str, i + 2);
        }
        return res;
    }

    // 第一个数字如果是3-9，就只能是个位数才能解析
    return process(str, i + 1);
}

    # 动态规划

    public static int dpWays(String s) {
    if (s == null || s.length() == 0) {
        return 0;
    }
    char[] str = s.toCharArray();
    int N = str.length;
    int[] dp = new int[N + 1];
    dp[N] = 1;
    for (int i = N - 1; i >= 0; i--) {
        if (str[i] == '0') {
            dp[i] = 0;
        }

        if (str[i] == '1') {
            dp[i] = dp[i + 1];
            if (i + 1 < str.length) {
                dp[i] += dp[i + 2];
            }
        }

        if (str[i] == '2') {
            dp[i] = dp[i + 1];
            // 第二个数字超过6就不能转化了
            if (i + 1 < str.length && (str[i + 1] >= '0' && str[i + 1] <= '6')) {
                dp[i] += dp[i + 2];
            }
        }
    }
    return dp[0];
}

    # 类型2 背包问题

    给定两个长度都为N的数组weights和values weights[i]和values[i]分别代表i号物品的重量和价值 给定一个正数bag，表示一个载重bag的袋子 你装的物品不能超过这个重量 返回你能装下最多的价值是多少

    # 暴力递归一

    public static int getMaxValue(int[] w, int[] v, int bag) {
    return process(w, v, 0, 0, bag);
}

// 从index开始往后的最大价值
private static int process(int[] w, int[] v, int index, int alreadyW, int bag) {
    // 超重
    if (alreadyW > bag) {
        return -1;
    }

    // 选到最后了 选完了
    if (index == w.length) {
        return 0;
    }

    // 当前index位置物品不加入背包，index+1开始往后得到的最大价值
    int p1 = process(w, v, index + 1, alreadyW, bag);
    // 当前index位置物品加入背包后，index+1开始往后得到的最大价值
    int p2next = process(w, v, index + 1, alreadyW + w[index], bag);
    int p2 = -1;
    if (p2next != -1) { // 表示当前index位置物品加入背包后不超重，可得到一个最大值
        p2 = v[index] + p2next;
    }
    return Math.max(p1, p2);
}

    # 暴力递归二 经典

    public static int maxValue(int[] w, int[] v, int bag) {
    return process(w, v, 0, bag);
}

// 背包中只剩下rest大小的空间了
private static int process(int[] w, int[] v, int index, int rest) {
    if (rest <= 0) { // base case 1
        return 0;
    }
    if (index == w.length) { // base case 2
        return 0;
    }

    // 当前index位置物品不加入背包，index+1开始往后得到的最大价值
    int p1 = process(w, v, index + 1, rest);
    int p2 = Integer.MIN_VALUE;
    if (rest >= w[index]) { // 表示当前index位置物品加入背包后不超重，可得到一个最大值
        p2 = v[index] + process(w, v, index + 1, rest - w[index]);
    }
    return Math.max(p1, p2);
}

    # 动态规划

    根据暴力递归二解法改动态递归

    public static int dpWay(int[] w, int[] v, int bag) {
    int N = w.length;
    int[][] dp = new int[N + 1][bag + 1];
    for (int index = N - 1; index >= 0; index--) {
        for (int rest = 1; rest <= bag; rest++) {
            dp[index][rest] = dp[index + 1][rest];
            if (rest >= w[index]) {
                dp[index][rest] = Math.max(dp[index][rest], v[index] + dp[index + 1][rest - w[index]]);
            }
        }
    }
    return dp[0][bag];
}

    # 例题一 纸牌博弈

    给定一个整型数组arr， 代表数值不同的纸牌排成一条线 玩家A和玩家B依次拿走每张纸牌 规定玩家A先拿，玩家B后拿 但是每个玩家每次只能拿走最左或最右的纸牌 玩家A和玩家B都绝顶聪明 请返回最后获胜者的分数

    # 暴力递归

    public static int win1(int[] arr) {
    if (arr == null || arr.length == 0) {
        return 0;
    }
    // 玩家A f(arr, 0, arr.length - 1)
    // 玩家B s(arr, 0, arr.length - 1)
    // 可能是A赢 也可能是B赢
    return Math.max(f(arr, 0, arr.length - 1), s(arr, 0, arr.length - 1));
}

// 先手
private static int f(int[] arr, int l, int r) {
    if (l == r) {
        return arr[l];
    }
    return Math.max(arr[l] + s(arr, l + 1, r), arr[r] + s(arr, l, r - 1));
}

// 后手
private static int s(int[] arr, int l, int r) {
    if (l == r) {
        return 0;
    }
    // 对手挑完，到后手的自己时候，对手一定给我方是最差的
    return Math.min(f(arr, l + 1, r), f(arr, l, r - 1));
}

    # dp

    public static int dpWay(int[] arr) {
    if (arr == null || arr.length == 0) {
        return 0;
    }
    int[][] f = new int[arr.length][arr.length];
    int[][] s = new int[arr.length][arr.length];
    for (int j = 0; j < arr.length; j++) {
        f[j][j] = arr[j];
        for (int i = j - 1; i >= 0; i--) {
            f[i][j] = Math.max(arr[i] + s[i + 1][j], arr[j] + s[i][j - 1]);
            s[i][j] = Math.min(f[i + 1][j], f[i][j - 1]);
        }
    }
    return Math.max(f[0][arr.length - 1], s[0][arr.length - 1]);
}

    # 题目二

    假设有排成一行的N个位置，记为1~N，N一定是大于或等于2的 开始时，机器人在其中的M位置上（M一定是1~N中的一个） 如果机器人来到1位置，那么下一步只能往右来到2位置 如果机器人来到N位置，那么下一步只能往左来到N-1位置 如果机器人来到中间位置，那么下一步可以往左走或者往右走 规定机器人必须走K步，最终能来到P位置（P也是1~N中的一个）的方法有多少种 给定四个参数N、M、K、P，返回方法数

    # 暴力递归

    public static int ways1(int N, int M, int K, int P) {
    // 参数无效直接返回0
    if (N < 2 || K < 1 || M < 1 || M > N || P < 1 || P > N) {
        return 0;
    }
    // 总共N个位置，从M点出发，还剩K步，返回最终能到达到P的方法数
    return walk(N, M, K, P);
}

/**
  * 只能在1-N这些位置上移动，当前在cur位置，走完rest步之后，停在P位置的方法数
  *
  * @param N    位置为1-N，固定参数
  * @param cur  当前在cur位置，可变参数
  * @param rest 还剩rest步没走，可变参数
  * @param p    最终目标位置，固定参数
  * @return 返回所有可行的走法
  */
private static int walk(int N, int cur, int rest, int p) {
    // 如果没有剩余步数，那么当前cur位置就是最后的位置
    // 如果cur在p位置上，就说明这种走法是正确的
    // 如果cur不在p位置上，就说明这种走法是不正确的
    if (rest == 0) {
        return cur == p ? 1 : 0;
    }

    // 按照规则，在1位置，只能往右走
    if (cur == 1) {
        return walk(N, 2, rest - 1, p);
    }

    // 如果在最后一个位置，只能往左走
    if (cur == N) {
        return walk(N, cur - 1, rest - 1, p);
    }

    // 如果在中间位置，既可以往左，也可以往右走
    return walk(N, cur + 1, rest - 1, p) + walk(N, cur - 1, rest - 1, p);
}

    # 记忆化搜索

    public static int ways2(int N, int M, int K, int P) {
    // 参数无效直接返回0
    if (N < 2 || K < 1 || M < 1 || M > N || P < 1 || P > N) {
        return 0;
    }
    int[][] memo = new int[N + 1][K + 1];
    for (int row = 0; row <= N; row++) {
        for (int col = 0; col <= K; col++) {
            memo[row][col] = -1;
        }
    }
    // 总共N个位置，从M点出发，还剩K步，返回最终能到达到P的方法数
    return walk2(N, M, K, P, memo);
}

/**
  * 只能在1-N这些位置上移动，当前在cur位置，走完rest步之后，停在P位置的方法数
  *
  * @param N    位置为1-N，固定参数
  * @param cur  当前在cur位置，可变参数
  * @param rest 还剩rest步没走，可变参数
  * @param p    最终目标位置，固定参数
  * @param memo 记录之前尝试的走法
  * @return 返回所有可行的走法
  */
private static int walk2(int N, int cur, int rest, int p, int[][] memo) {
    if (memo[cur][rest] != -1) {
        return memo[cur][rest];
    }
    // 如果没有剩余步数，那么当前cur位置就是最后的位置
    // 如果cur在p位置上，就说明这种走法是正确的
    // 如果cur不在p位置上，就说明这种走法是不正确的
    if (rest == 0) {
        memo[cur][rest] = cur == p ? 1 : 0;
        return memo[cur][rest];
    }

    // 按照规则，在1位置，只能往右走
    if (cur == 1) {
        memo[cur][rest] = walk2(N, 2, rest - 1, p, memo);
        return memo[cur][rest];
    }

    // 如果在最后一个位置，只能往左走
    if (cur == N) {
        memo[cur][rest] = walk2(N, cur - 1, rest - 1, p, memo);
        return memo[cur][rest];
    }

    // 如果在中间位置，既可以往左，也可以往右走
    memo[cur][rest] = walk2(N, cur + 1, rest - 1, p, memo) + walk2(N, cur - 1, rest - 1, p, memo);
    return memo[cur][rest];
}

    # dp

    public static int ways2(int N, int M, int K, int P) {
    // 参数无效直接返回0
    if (N < 2 || K < 1 || M < 1 || M > N || P < 1 || P > N) {
        return 0;
    }
    int[][] dp = new int[K + 1][N + 1];
    dp[0][P] = 1;
    for (int i = 1; i <= K; i++) {
        for (int j = 1; j <= N; j++) {
            if (j == 1) {
                dp[i][j] = dp[i - 1][2];
            } else if (j == N) {
                dp[i][j] = dp[i - 1][N - 1];
            } else {
                dp[i][j] = dp[i - 1][j - 1] + dp[i - 1][j + 1];
            }
        }
    }
    return dp[K][M];
}

    # 多样本位置全对应的尝试模型

    # 例题一 两个字符串的最长公共子序列问题 (opens new window)

    一个样本做行，一个样本做列

    public static int lcs(char[] str1, char[] str2) {
  int[][] dp = new int[str1.length][str2.length];
  // base case
  dp[0][0] = str1[0] == str2[0] ? 1 : 0;
  for (int i = 1; i < str1.length; i++) {
    dp[i][0] = Math.max(dp[i - 1][0], str1[i] == str2[0] ? 1 : 0);
  }
  for (int j = 1; j < str2.length; j++) {
    dp[0][j] = Math.max(dp[0][j - 1], str1[0] == str2[j] ? 1 : 0);
  }

  // 一共有三种可能
  // 1. 不以两个串当前尾字符结尾 dp[i - 1][j - 1] -> 该情况可以省略，因为情况二的结果必然不比情况1小，可以自己手动画出dp表观察
  // 2. 以两个串中的某一个串的尾字符结尾 dp[i][j - 1] 或者 dp[i - 1][j]
  // 3. 如果两个串当前尾字符相等，也可能以两个串当前尾字符结尾 dp[i - 1][j - 1] + 1
  for (int i = 1; i < str1.length; i++) {
    for (int j = 1; j < str2.length; j++) {
      // 以其中某一个字符串的当前位置结尾
      dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
      if (str1[i] == str2[j]) { // 以两个串的当前位置结尾
        dp[i][j] = Math.max(dp[i][j], dp[i - 1][j - 1] + 1);
      }
    }
  }
  return dp[str1.length - 1][str2.length - 1];
}

    # 寻找业务限制的尝试模型

    # 例题一

    给定一个数组，代表每个人喝完咖啡准备刷杯子的时间 数组是有序的 只有一台咖啡机，一次只能洗一个杯子，时间耗费a，洗完才能洗下一杯 每个咖啡杯也可以自己挥发干净，时间耗费b，咖啡杯可以并行挥发 返回让所有咖啡杯变干净的最早完成时间 三个参数：int[] drinks、int a、int b

    # 暴力递归

    public static int minTime(int[] drinks, int a, int b) {
    if (a >= b) { // 如果挥发时间比洗的时间短，就全挥发就完事了
        return drinks[drinks.length - 1] + b;
    }
    return process(drinks, a, b, 0, 0);
}
/**
  * drinks[..., index - 1]的杯子都已经干净
  * drinks[index, ...] 还没洗
  *
  * @param drinks   每个员工喝完的时间，固定变量
  * @param a        洗一杯的时间，固定变量
  * @param b        挥发干净的时间，固定变量
  * @param index    还没有干净的杯子的首位
  * @param washLine 洗杯子的机器何时可用
  * @return 杯子全部干净的最少时间点
  */
public static int process(int[] drinks, int a, int b, int index, int washLine) {
    if (index == drinks.length - 1) {
        return Math.min(
                Math.max(washLine, drinks[index]) + a, // 用咖啡机洗
                drinks[index] + b // 挥发
        );
    }
    // 当前咖啡杯决定用咖啡机洗
    // wash是我当前的咖啡杯，洗完的时间
    int wash = Math.max(washLine, drinks[index]) + a;
    int next1 = process(drinks, a, b, index + 1, wash);
    int p1 = Math.max(wash, next1); // 谁大，谁就是最后的时刻

    // 当前咖啡杯决定挥发干净
    int dry = drinks[index] + b;
    int next2 = process(drinks, a, b, index + 1, washLine);
    int p2 = Math.max(dry, next2); // 谁大，谁就是最后的时刻

    return Math.min(p1, p2);
}

    # dp

    /**
* @param drinks 每个员工喝完的时间
* @param a      洗一杯的时间
* @param b      挥发干净的时间
* @return 杯子全部干净的最少时间点
*/
public static int dp(int[] drinks, int a, int b) {
    if (a >= b) { // 如果挥发时间比洗的时间短，就全挥发就完事了
        return drinks[drinks.length - 1] + b;
    }

    int limit = 0; // 咖啡机什么时候可用
    // 这边将得到最极限情况下咖啡机的可用时刻
    // 即咖啡机一旦可用，下一个杯子就进去洗，不停地工作
    for (int i = 0; i < drinks.length - 1; i++) {
        limit = Math.max(limit, drinks[i]) + a;
    }

    int[][] dp = new int[drinks.length][limit + 1];

    // base case
    for (int washLine = 0; washLine <= limit; washLine++) {
        dp[drinks.length - 1][washLine] = Math.min(
                Math.max(washLine, drinks[drinks.length - 1]) + a,
                drinks[drinks.length - 1] + b
        );
    }

    for (int index = drinks.length - 2; index >= 0; index--) {
        for (int washLine = 0; washLine <= limit; washLine++) {
            int p1 = Integer.MAX_VALUE;
            int wash = Math.max(washLine, drinks[index]) + a;
            if (wash <= limit) {
                p1 = Math.max(wash, dp[index + 1][wash]);
            }

            int p2 = Math.max(drinks[index] + b, dp[index + 1][washLine]);

            dp[index][washLine] = Math.min(p1, p2);
        }
    }

    return dp[0][0];
}
