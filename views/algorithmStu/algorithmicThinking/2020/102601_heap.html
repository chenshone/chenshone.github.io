    堆结构

    CHENSHONE 2020/10/26 算法思维

    1. 堆结构就是用数组实现的完全二叉树结构
    2. 优先级队列就是堆结构

    # 一些概念设计

    # 数组与二叉树

    上面提到堆就是用数组模拟二叉树结构。所以，关于数组对应的二叉树位置有如下规律：

    对于某一位置i

    1. 左孩子对应的位置是2*i+1
    2. 右孩子对应的位置是2*i+2
    3. 父节点对应的位置是(i-1)/2

    # 大根堆和小根堆

    堆分为大根堆和小根堆，所谓大/小，就是说，对于堆中的某一节点a，它的所有子节点都比它大/小。

    # heapInsert

    每次push一个新元素时候的操作，将新元素添加到最后的位置，然后进行heapInsert操作，让其每次都与其所在位置的父节点比较，如果比父节点大，就与父节点交换位置，循环往复，直到比父节点或者到堆顶即停止操作，push完成。

    # heapify

    每次pop的操作，将最末尾的数与最大数(0位置)交换，堆大小缩减1个空间，然后将刚交换到0位置的那个数与其左右子节点中较大的数比较，如果当前位置没有左右子节点中较大的数大，就与其交换位置，循环往复，直到比左右子节点中较大的还要大或者到达堆底即停止操作，pop完成。

    # 自上而下与自下而上

    1. 自上而下：对应heapInsert，即从数组开头开始依次整理成堆结构
    2. 自下而上：对应heapify，即从数组结尾开始依次整理成堆结构

    # 堆结构代码（大根堆）

    public class MaxHeap {
    private int heapSize;
    private final int limit;
    private int[] heap;

    public MaxHeap(int limit) {
        heap = new int[limit];
        this.limit = limit;
        heapSize = 0;
    }

    public boolean isEmpty() {
        return heapSize == 0;
    }

    public boolean isFull() {
        return heapSize == limit;
    }

    public void push(int value) {
        if (heapSize == limit) {
            throw new RuntimeException("heap is full");
        }
        heap[heapSize] = value;
        heapInsert(heap, heapSize++);
    }

    //    将要pop的的头部与尾部swap，然后将剩下的heapify
    public int pop() {
        int ans = heap[0];
        swap(heap, 0, --heapSize);
        heapify(heap, 0, heapSize);
        return ans;
    }


    private void swap(int[] heap, int a, int b) {
        int tmp = heap[a];
        heap[a] = heap[b];
        heap[b] = tmp;
    }

    private void heapInsert(int[] heap, int index) {
        while (heap[index] > heap[(index - 1) / 2]) {
            swap(heap, index, (index - 1) / 2);
            index = (index - 1) / 2;
        }
    }

    private void heapify(int[] heap, int index, int heapSize) {
        int left = index * 2 + 1;
//        当前有子节点（子节点不越界）
        while (left < heapSize) {
//            左右孩子比较，哪个大，就拿谁进行比较
            int largest = (left + 1 < heapSize && heap[left + 1] > heap[left]) ? left + 1 : left;
            largest = heap[largest] > heap[index] ? largest : index;
            if (largest == index) {
                break;
            }
            swap(heap, largest, index);
            index = largest;
            left = index * 2 + 1;
        }
    }
}

    # 堆排序 不稳定

    # 思路

    将整个数组变成大根堆，然后将最大值（堆顶）与末尾交换，然后向左缩小一个位置，然后调整堆结构，循环往复即可。

    # 代码

    public class HeapSort {
    public static void heapSort(int[] arr) {
        if (arr == null || arr.length < 2) {
            return;
        }

//        自上而下 O(N*logN)
        /*for (int i = 0; i < arr.length; i++) {
            heapInsert(arr, i);
        }*/

//        自下而上 O(N)
        for (int i = arr.length - 1; i >= 0; i--) {
            heapify(arr, i, arr.length);
        }

        int heapSize = arr.length;
        swap(arr, 0, --heapSize);

        while (heapSize > 0) {
            heapify(arr, 0, heapSize);
            swap(arr, 0, --heapSize);
        }
    }

    private static void heapify(int[] arr, int i, int heapSize) {
        int left = i * 2 + 1;
//        当前有子节点（子节点不越界）
        while (left < heapSize) {
//            左右孩子比较，哪个大，就拿谁进行比较
            int largest = (left + 1 < heapSize && arr[left + 1] > arr[left]) ? left + 1 : left;
            largest = arr[largest] > arr[i] ? largest : i;
            if (largest == i) {
                break;
            }
            swap(arr, largest, i);
            i = largest;
            left = i * 2 + 1;
        }
    }

    private static void heapInsert(int[] arr, int i) {
        while (arr[i] > arr[(i - 1) / 2]) {
            swap(arr, i, (i - 1) / 2);
            i = (i - 1) / 2;
        }
    }

    private static void swap(int[] arr, int a, int b) {
        int tmp = arr[a];
        arr[a] = arr[b];
        arr[a] = tmp;
    }
}
