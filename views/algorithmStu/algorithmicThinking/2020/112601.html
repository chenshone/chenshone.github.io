    矩阵处理

    CHENSHONE 2020/11/26 算法思维

    核心技巧

    找到coding上的宏观调度

    # zigzag打印矩阵

    按照下面的轨迹打印矩阵 题意

    # 思路

    站在更宏观的角度看

    一个二维矩阵，row和col，设a在col上，b在row上，

    起初他们都在0位置，

    a每次都往右走，直到不能往右就往下，

    b每次都往下走，直到不能往下就往右

    这样每次a和b都能连成一条斜线，这样只要遍历每条斜线，看好每条斜线是从哪头开始遍历即可

    # 代码

    public static void printMatrixZigZag(int[][] matrix) {
    int Ar = 0; // point a row
    int Ac = 0; // point a col
    int Br = 0; // point b row
    int Bc = 0; // point b col
    int Endr = matrix.length - 1; // row end
    int Endc = matrix[0].length - 1; // col end
    boolean fromUp = false; // 是否从右上角往左下角打印
    while (Ar != Endr + 1) {
        printLevel(matrix, Ar, Ac, Br, Bc, fromUp);
        Ar = Ac == Endc ? Ar + 1 : Ar;
        Ac = Ac == Endc ? Ac : Ac + 1;

        // 不要更换二者位置
        Bc = Br == Endr ? Bc + 1 : Bc;
        Br = Br == Endr ? Br : Br + 1;
        fromUp = !fromUp;
    }
    System.out.println();
}

private static void printLevel(int[][] matrix, int aR, int aC, int bR, int bC, boolean fromUp) {
    if (fromUp) {
        while (aR != bR + 1) {
            System.out.println(matrix[aR++][aC--] + " ");
        }
    } else {
        while (bR != aR - 1) {
            System.out.println(matrix[bR--][bC++] + " ");
        }
    }
}

    # 转圈打印矩阵

    要求：leetcode 顺时针打印矩阵 (opens new window)

    # 思路

    可以把矩阵分层，最外围是一层，向内依次为第二层、第三层。。。

    每层都标记出左上角和右下角两个坐标，四个标记，

    一圈又可以分为四个方向：向右、向下、向左、向上，每个方向都依次打印到边界（不要打印边界）

    # 代码

    public static void spiralOrderPrint(int[][] matrix) {
    // 左上角坐标
    int tR = 0;
    int tC = 0;
    // 右下角坐标
    int dR = matrix.length - 1;
    int dC = matrix[0].length - 1;
    while (tR <= dR && tC <= dC) {
        printEdge(matrix, tR++, tC++, dR--, dC--);
    }
}

private static void printEdge(int[][] matrix, int tR, int tC, int dR, int dC) {
    if (tR == dR) {
        for (int i = tC; i <= dC; i++) {
            System.out.println(matrix[tR][i] + " ");
        }
    } else if (tC == dC) {
        for (int i = tR; i <= dR; i++) {
            System.out.println(matrix[i][tC] + " ");
        }
    } else {
        int curR = tR;
        int curC = tC;
        while (curC != dC) { // left top -> right top
            System.out.println(matrix[tR][curC++] + "");
        }
        while (curR != dR) { // right top -> right bottom
            System.out.println(matrix[curR++][dC] + "");
        }
        while (curC != tC) { // right bottom -> left bottom
            System.out.println(matrix[dR][curC--] + " ");
        }
        while (curR != tR) { // left bottom -> left top
            System.out.println(matrix[curR--][tC] + " ");
        }
    }
}

    # 原地旋转正方形矩阵（顺时针）

    # 思路

    跟上一个一样

    # 代码

    public static void rotate(int[][] matrix) {
    int a = 0;
    int b = 0;
    int c = matrix.length - 1;
    int d = matrix[0].length - 1;
    while (a < c) {
        rotateEdge(matrix, a++, b++, c--, d--);
    }
}

private static void rotateEdge(int[][] matrix, int a, int b, int c, int d) {
    int tmp = 0;
    for (int i = 0; i < d - b; i++) {
        tmp = matrix[a][b + i];
        matrix[a][b + i] = matrix[c - i][b];
        matrix[c - i][b] = matrix[c][d - i];
        matrix[c][d - i] = matrix[a + i][d];
        matrix[a + i][d] = tmp;
    }
}

public static void printMatrix(int[][] matrix) {
    for (int i = 0; i < matrix.length; i++) {
        for (int j = 0; j < matrix[0].length; j++) {
            System.out.println(matrix[i][j] + " ");
        }
        System.out.println();
    }
}
