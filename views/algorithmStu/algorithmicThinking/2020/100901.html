    线段树

    CHENSHONE 2020/10/9 算法思维

    线段树就是区间修改树，就是用来修改一个区间内的元素的

    适用范围

    query操作需要满足：父节点的信息可以由左孩子和右孩子的信息直接加工得到，不需要管具体左树和右树的状态。

    本篇文章中数组下标均从1开始

    # 所用场景

    对于一个长度为n的数组arr，现要求实现3个方法：

    1. void add(L, R, V, arr): 将arr数组中[L, R]的每个元素都加V
    2. void update(L, R, V, arr): 将arr数组中[L, R]的每个元素都更改为V
    3. int query(L, R, arr): 返回arr数组中[L, R]的元素和

    要求

    每个方法的复杂度为O(logN)

    # 线段树功能分析

    首先，对于这样的需求，我们需要事先拿到目标数组所有可能的子区间的信息。

    对于理想型数组，即长度为2^n的数组，可以采用二分法

    如上图所示，一个长度为8的数组，需要15(2 * 8 - 1)个空间存放信息，这样的话，就可以通过组合的方式拿到所有子区间的信息了。

    对于非长度为2^n的数组，比如长度为6的数组，可以通过补0的方式将其扩充为长度为2^n的数组。

    结论

    开辟 4n 的空间是完全足够的

    # 线段树信息建立

    使用数组的方式存储线段树信息

    如上图所示，线段树需要的信息是区间和，要得到线段树对应的元素的value，对应的某个元素i，它的左孩子是2*i，右孩子是2*i+1。

    # 代码演示

    public class SegmentTree {
    private int MAXN;
    private int[] arr; // 从下标1开始
    private int[] sum; // 模拟线段树维护区间和

    public SegmentTree(int[] origin) {
        MAXN = origin.length + 1;
        arr = new int[MAXN]; // arr[0] 不使用 从arr[1]开始使用
        for (int i = 1; i < MAXN; i++) {
            arr[i] = origin[i - 1];
        }
        sum = new int[MAXN << 2]; // 用来支持，某一个范围的累加和信息
    }

    // 在初始化阶段，先把sum数组填好
    // 在arr[1 - r]范围上，去build ，1 - n
    // rt: 这个范围在sum中的下标
    public void build(int l, int r, int rt) {
        if (l == r) {
            sum[rt] = arr[l];
            return;
        }
        int mid = (l + r) >> 1;
        build(l, mid, rt << 1);
        build(mid + 1, r, rt << 1 | 1);
        pushUp(rt);
    }

    private void pushUp(int rt) {
        sum[rt] = sum[rt << 1] + sum[rt << 1 | 1];
    }
}

    # 累加操作(add方法)

    假设现在有个add操作，在区间[1, 6]上每个元素都加3，那么，对于线段树，[1, 4]和[5, 8], [1, 4]是完全包含，[5, 8]中再二分到[5, 6]是完全包含的，就在lazy数组对于的位置(和线段树数组一样的位置)记录加3信息即可。

    在该add操作后，如果还有个add操作，是在[3, 4]区间的，那么对于之前[1, 4]的lazy，需要下放到[1, 2]和[3, 4]，然后自己的lazy位置清零，再重复上面的步骤。

    # 代码演示

    public class SegmentTree {
    // ...
    private int[] lazy; // 累加懒惰标记

    public SegmentTree(int[] origin) {
        // ...
        lazy = new int[MAXN << 2]; // 某一个范围没有往下传递的累加和
    }

    private void pushUp(int rt) {
        sum[rt] = sum[rt << 1] + sum[rt << 1 | 1];
    }

    // ln表示左子树元素节点个数，rn表示右子树元素节点个数
    private void pushDown(int rt, int ln, int rn) {
        /* 当前节点之前有lazy任务才下发，没有就跳过 */
        if (lazy[rt] != 0) {
            lazy[rt << 1] += lazy[rt];
            sum[rt << 1] += lazy[rt] * ln;

            lazy[rt << 1 | 1] += lazy[rt];
            sum[rt << 1 | 1] += lazy[rt] * rn;

            lazy[rt] = 0;
        }
    }

    // L R -> 任务范围，所有的值累加上C
    // l, r -> 表达的范围
    // rt 去哪找l, r范围上的信息
    public void add(int L, int R, int C,
                    int l, int r,
                    int rt) {
        // 任务的范围彻底覆盖了当前表达的范围
        if (L <= l && r <= R) {
            sum[rt] += C * (r - l + 1);
            lazy[rt] += C;
            return;
        }
        // 任务并没有把[l, r]全包住
        // 要把任务往下发
        // 任务 L, R 没有把本身表达范围l, r 彻底包住
        int mid = (l + r) >> 1;
        // 下发之前的lazy add任务
        pushDown(rt, mid - l + 1, r - mid);
        // 左孩子是否需要接到任务
        if (L <= mid) {
            add(L, R, C, l, mid, rt << 1);
        }
        // 右孩子是否需要接到任务
        if (mid < R) {
            add(L, R, C, mid + 1, r, rt << 1 | 1);
        }
        pushUp(rt);
    }
}

    # 更新操作(update方法)

    思路跟累加大体一致，区别在于pushDown方法增加了update的逻辑

    # 代码演示

    public class SegmentTree {
    // ...
    private int[] change; // 更新的值
    private boolean[] update; // 更新懒惰标记

    public SegmentTree(int[] origin) {
        // ...
        change = new int[MAXN << 2]; // 某一个范围有没有更新操作的懒任务，和下面的update数组搭配使用，存放对应的懒任务的信息
        update = new boolean[MAXN << 2]; // 某一个范围更新任务,如果有update的懒任务就为true
    }

    private void pushUp(int rt) {
        sum[rt] = sum[rt << 1] + sum[rt << 1 | 1];
    }

    // 之前的所有懒增加和懒更新，从父范围发给左右两个子范围的分发策略
    // ln表示左子树元素节点个数，rn表示右子树元素节点个数
    private void pushDown(int rt, int ln, int rn) {
        /* 当前节点之前有lazy任务才下发，没有就跳过 */
        if (update[rt]) {
            update[rt << 1] = true;
            update[rt << 1 | 1] = true;
            change[rt << 1] = change[rt];
            change[rt << 1 | 1] = change[rt];
            lazy[rt << 1] = 0;
            lazy[rt << 1 | 1] = 0;
            sum[rt << 1] = change[rt] * ln;
            sum[rt << 1 | 1] = change[rt] * rn;
            update[rt] = false;
        }
        /* 对于一系列的操作，先进行的add必然会被后进行的update覆盖
         * 但是，update之后发生的add操作不应该被update覆盖
         * 所以，对于先update，后add的一系列懒任务
         * 在其下发的过程中，下发完update后，也要对lazy add任务进行下发
         * */
        if (lazy[rt] != 0) {
            lazy[rt << 1] += lazy[rt];
            sum[rt << 1] += lazy[rt] * ln;

            lazy[rt << 1 | 1] += lazy[rt];
            sum[rt << 1 | 1] += lazy[rt] * rn;

            lazy[rt] = 0;
        }
    }

    // L R -> 任务范围，所有的值更新成C
    // l, r -> 表达的范围
    // rt 去哪找l, r范围上的信息
    public void update(int L, int R, int C,
                       int l, int r,
                       int rt) {
        if (L <= l && r <= R) {
            update[rt] = true;
            change[rt] = C;
            sum[rt] = C * (r - l + 1);
            lazy[rt] = 0;
            return;
        }
        /*当前任务无法懒更新，必须下发*/
        int mid = (l + r) >> 1;
        pushDown(rt, mid - l + 1, r - mid);
        if (L <= mid) {
            update(L, R, C, l, mid, rt << 1);
        }
        if (mid < R) {
            update(L, R, C, mid + 1, r, rt << 1 | 1);
        }
        pushUp(rt);
    }
}

    # 查询区间和(query方法)

    思路同上,能拦住就直接return，不能就下发，然后继续递归。

    # 代码演示

    public long query(int L, int R, int l, int r, int rt) {
    if (L <= l && r <= R) {
        return sum[rt];
    }
    int mid = (l + r) >> 1;
    pushDown(rt, mid - l + 1, r - mid);
    long ans = 0;
    if (L <= mid) {
        ans += query(L, R, l, mid, rt);
    }
    if (mid < R) {
        ans += query(L, R, mid + 1, r, rt);
    }
    return ans;
}

    public class SegmentTree {
    private int MAXN;
    private int[] arr; // 从下标1开始
    private int[] sum; // 模拟线段树维护区间和
    private int[] lazy; // 累加懒惰标记
    private int[] change; // 更新的值
    private boolean[] update; // 更新懒惰标记

    public SegmentTree(int[] origin) {
        MAXN = origin.length + 1;
        arr = new int[MAXN]; // arr[0] 不使用 从arr[1]开始使用
        for (int i = 1; i < MAXN; i++) {
            arr[i] = origin[i - 1];
        }
        sum = new int[MAXN << 2]; // 用来支持，某一个范围的累加和信息
        lazy = new int[MAXN << 2]; // 某一个范围没有往下传递的累加和
        change = new int[MAXN << 2]; // 某一个范围有没有更新操作的懒任务，和下面的update数组搭配使用，存放对应的懒任务的信息

    }

    private void pushUp(int rt) {
        sum[rt] = sum[rt << 1] + sum[rt << 1 | 1];

            update[rt << 1] = true;
            update[rt << 1 | 1] = true;
            change[rt << 1] = change[rt];
            change[rt << 1 | 1] = change[rt];
            lazy[rt << 1] = 0;
            lazy[rt << 1 | 1] = 0;
            sum[rt << 1] = change[rt] * ln;
            sum[rt << 1 | 1] = change[rt] * rn;
            update[rt] = false;
        }

         * 但是，update之后发生的add操作不应该被update覆盖
         * 所以，对于先update，后add的一系列懒任务
         * 在其下发的过程中，下发完update后，也要对lazy add任务进行下发
         * */

            lazy[rt << 1] += lazy[rt];

            lazy[rt << 1 | 1] += lazy[rt];
            sum[rt << 1 | 1] += lazy[rt] * rn;

            lazy[rt] = 0;
        }
    }

    // 在初始化阶段，先把sum数组填好
    // 在arr[1 - r]范围上，去build ，1 - n
    // rt: 这个范围在sum中的下标
    public void build(int l, int r, int rt) {
        if (l == r) {
            sum[rt] = arr[l];
            return;
        }
        int mid = (l + r) >> 1;
        build(l, mid, rt << 1);
        build(mid + 1, r, rt << 1 | 1);
        pushUp(rt);
    }


    // L R -> 任务范围，所有的值累加上C
    // l, r -> 表达的范围
    // rt 去哪找l, r范围上的信息
    public void add(int L, int R, int C,
                    int l, int r,
                    int rt) {
        // 任务的范围彻底覆盖了当前表达的范围
        if (L <= l && r <= R) {
            sum[rt] += C * (r - l + 1);
            lazy[rt] += C;
            return;
        }
        // 任务并没有把[l, r]全包住
        // 要把任务往下发
        // 任务 L, R 没有把本身表达范围l, r 彻底包住
        int mid = (l + r) >> 1;
        // 下发之前的lazy add任务
        pushDown(rt, mid - l + 1, r - mid);
        // 左孩子是否需要接到任务
        if (L <= mid) {
            add(L, R, C, l, mid, rt << 1);
        }
        // 右孩子是否需要接到任务
        if (mid < R) {
            add(L, R, C, mid + 1, r, rt << 1 | 1);
        }
        pushUp(rt);
    }

    // L R -> 任务范围，所有的值更新成C
    // l, r -> 表达的范围
    // rt 去哪找l, r范围上的信息
    public void update(int L, int R, int C,
                       int l, int r,
                       int rt) {
        if (L <= l && r <= R) {
            update[rt] = true;
            change[rt] = C;
            sum[rt] = C * (r - l + 1);
            lazy[rt] = 0;
            return;
        }
        /*当前任务无法懒更新，必须下发*/
        int mid = (l + r) >> 1;
        pushDown(rt, mid - l + 1, r - mid);
        if (L <= mid) {
            update(L, R, C, l, mid, rt << 1);
        }
        if (mid < R) {
            update(L, R, C, mid + 1, r, rt << 1 | 1);
        }
        pushUp(rt);
    }

    public long query(int L, int R, int l, int r, int rt) {
        if (L <= l && r <= R) {
            return sum[rt];
        }
        int mid = (l + r) >> 1;
        pushDown(rt, mid - l + 1, r - mid);
        long ans = 0;
        if (L <= mid) {
            ans += query(L, R, l, mid, rt);
        }
        if (mid < R) {
            ans += query(L, R, mid + 1, r, rt);
        }
        return ans;
    }
}

    # 应用

    想想一样标准的俄罗斯方块游戏，x轴是积木最终落下到底的轴线 下面是简化版

    1. 只会落下正方形积木
    2. [a, b] 表示 一个边长为b的正方形积木，积木左边缘沿着 x=a 这条线从上方掉落
    3. 认为x轴没有左右边界，可以无限放左右放
    4. 不会有哪一层会填满，也就是不会消除积木

    给定一个N*2的二维数组matrix，表示N个积木依次掉落 返回每一次掉落之后的最大高度

    public class FallingSquares {
    public static class SegmentTree {
        private int[] max;
        private int[] change;
        private boolean[] update;

        public SegmentTree(int size) {
            int N = size + 1;
            max = new int[N << 2];

            change = new int[N << 2];
            update = new boolean[N << 2];
        }

        private void pushUp(int rt) {
            max[rt] = Math.max(max[rt << 1], max[rt << 1 | 1]);
        }

        // ln表示左子树元素结点个数，rn表示右子树结点个数
        private void pushDown(int rt, int ln, int rn) {
            if (update[rt]) {
                update[rt << 1] = true;
                update[rt << 1 | 1] = true;
                change[rt << 1] = change[rt];
                change[rt << 1 | 1] = change[rt];
                max[rt << 1] = change[rt];
                max[rt << 1 | 1] = change[rt];
                update[rt] = false;
            }
        }

        public void update(int L, int R, int C, int l, int r, int rt) {
            if (L <= l && r <= R) {
                update[rt] = true;
                change[rt] = C;
                max[rt] = C;
                return;
            }
            int mid = (l + r) >> 1;
            pushDown(rt, mid - l + 1, r - mid);
            if (L <= mid) {
                update(L, R, C, l, mid, rt << 1);
            }
            if (R > mid) {
                update(L, R, C, mid + 1, r, rt << 1 | 1);
            }
            pushUp(rt);
        }

        public int query(int L, int R, int l, int r, int rt) {
            if (L <= l && r <= R) {
                return max[rt];
            }
            int mid = (l + r) >> 1;
            pushDown(rt, mid - l + 1, r - mid);
            int left = 0;
            int right = 0;
            if (L <= mid) {
                left = query(L, R, l, mid, rt << 1);
            }
            if (R > mid) {
                right = query(L, R, mid + 1, r, rt << 1 | 1);
            }
            return Math.max(left, right);
        }

    }

    // positions 只取最左边和最右边表示这块积木所占位置
    // [2,7] -> 2 , 8
    // [3, 10] -> 3, 12

    // 离散化
    //
    public HashMap<Integer, Integer> index(int[][] positions) {
        TreeSet<Integer> pos = new TreeSet<>(); // 有序表
        for (int[] arr : positions) {
            pos.add(arr[0]);
            pos.add(arr[0] + arr[1] - 1);
        }
        HashMap<Integer, Integer> map = new HashMap<>();
        int count = 0;
        for (Integer index : pos) {
            map.put(index, ++count);
        }
        return map;
    }

    public List<Integer> fallingSquares(int[][] positions) {
        HashMap<Integer, Integer> map = index(positions);
        // 100   -> 1    306 ->   2   403 -> 3
        // [100,403]   1~3
        int N = map.size(); // 1 ~ 	N
        SegmentTree segmentTree = new SegmentTree(N);
        int max = 0;
        List<Integer> res = new ArrayList<>();
        // 每落一个正方形，收集一下，所有东西组成的图像，最高高度是什么
        for (int[] arr : positions) {
            int L = map.get(arr[0]);
            int R = map.get(arr[0] + arr[1] - 1);
            int height = segmentTree.query(L, R, 1, N, 1) + arr[1];
            max = Math.max(max, height);
            res.add(max);
            segmentTree.update(L, R, height, 1, N, 1);
        }
        return res;
    }
