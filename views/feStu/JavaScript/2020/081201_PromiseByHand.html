    # 函数版本

    ;(function (window) {
  const PENDING = 'pending'
  const RESOLVED = 'resolved'
  const REJECTED = 'rejected'
  /**
   * 构造函数
   *
   * @param {*} executer 执行器函数(同步执行)
   */
  function Promise(executer) {
    const self = this
    self.status = PENDING // 指定promise当前的状态，初始值为pending
    self.data = undefined // 用于存储结果数据
    self.callbacks = [] // 每个元素的结构为 { onResolved() {}, onRejected() {}}

    function resolve(value) {
      // 如果当前状态不为pending，直接结束
      if (self.status !== PENDING) {
        return
      }

      // 将状态改为resolved
      self.status = RESOLVED

      //保存value数据
      self.data = value

      // 如果有待执行的callback函数，就立刻执行异步回调函数 onResolved
      if (self.callbacks.length > 0) {
        setTimeout(() => {
          // 将所有成功的回调放入队列中执行
          self.callbacks.forEach(({ onResolved }) => {
            onResolved(value)
          })
        }, 0)
      }
    }

    function reject(err) {
      // 如果当前状态不为pending，直接结束
      if (self.status !== PENDING) {
        return
      }

      // 将状态改为rejected
      self.status = REJECTED

      //保存err数据
      self.data = err

      // 如果有待执行的callback函数，就立刻执行异步回调函数 onRejected
      if (self.callbacks.length > 0) {
        setTimeout(() => {
          // 将所有成功的回调放入队列中执行
          self.callbacks.forEach(({ onRejected }) => {
            onRejected(err)
          })
        }, 0)
      }
    }

    // 立刻同步执行executer
    try {
      executer(resolve, reject)
    } catch (error) {
      // 如果执行器抛出异常，就进入rejected状态
      reject(error)
    }
  }

  /**
   * 实例方法
   * 指定成功和失败的回调函数
   * 返回一个新的promise对象
   * @param {*} onResolved 成功的回调
   * @param {*} onRejected 失败的回调
   */
  Promise.prototype.then = function (onResolved, onRejected) {
    const self = this

    onResolved =
      typeof onResolved === 'function' ? onResolved : (value) => value

    // 指定默认的失败回调(实现错误/异常传透的关键点)
    onRejected =
      typeof onRejected === 'function'
        ? onRejected
        : (err) => {
            throw err
          }

    return new Promise((resolve, reject) => {
      if (self.status === PENDING) {
        // 假设当前状态为pending，将回调函数保存起来
        self.callbacks.push({
          onResolved(value) {
            handle(onResolved)
          },
          onRejected(err) {
            handle(onRejected)
          },
        })
      } else if (self.status === RESOLVED) {
        setTimeout(() => {
          handle(onResolved)
        }, 0)
      } else if (self.status === REJECTED) {
        setTimeout(() => {
          handle(onRejected)
        }, 0)
      }

      /**
       * 调用指定的回调函数处理
       *
       * @param {*} callback
       */
      function handle(callback) {
        /**
         * 1. 抛出异常，return的promise为失败结果
         * 2. 回调返回的结果为promise，return的promise为成功结果
         * 3. 回调返回的结果为非promise，return的promise结果就是这个结果
         */
        try {
          const result = callback(self.data)
          if (result instanceof Promise) {
            /* result.then(
                (value) => {
                  resolve(value) // 当result成功时，return的promise也成功
                },
                (err) => {
                  reject(err) // 当result失败时，return的promise也失败
                }
              ) */
            result.then(resolve, reject)
          } else {
            resolve(result)
          }
        } catch (error) {
          reject(error)
        }
      }
    })
  }

  /**
   * 实例方法
   * 指定失败的回调函数
   * 返回一个新的promise对象
   * @param {*} onRejected 失败的回调
   */
  Promise.prototype.catch = function (onRejected) {
    return this.then(undefined, onRejected)
  }

  /**
   * 静态方法
   * 返回一个指定结果的成功的promise
   * @param {*} value
   */
  Promise.resolve = function (value) {
    // 返回一个成功/失败的promise
    return new Promise((resolve, reject) => {
      if (value instanceof Promise) {
        value.then(resolve, reject)
      } else {
        resolve(value)
      }
    })
  }

  /**
   * 静态方法
   * 返回一个指定结果的失败的promise

   * @param {*} err
   */
  Promise.reject = function (err) {
    return new Promise((resolve, reject) => {
      reject(err)
    })
  }

  /**
   * 静态方法
   * 返回一个promise对象，只有当所有promise都成功时才会成功，否则失败
   * @param {*} promises
   */
  Promise.all = function (promises) {
    const len = promises.length
    const results = new Array(len) // 保存所有成功的结果
    let counts = 0 // 记录成功的结果的个数
    return new Promise((resolve, reject) => {
      // 遍历promises，获取每个promise的结果
      promises.forEach((p, index) => {
        Promise.resolve(p).then(
          (value) => {
            counts++

            // 将结构保存到数组中
            results[index] = value

            // 如果全部成功，则resolve出去
            if (counts === len) {
              resolve(results)
            }
          },
          (err) => {
            // 只要有一个promise失败，整个就失败，直接reject出去
            reject(err)
          }
        )
      })
    })
  }

  /**
   * 静态方法
   * 返回一个promise对象，其结果有第一个成功的promise决定，否则就失败
   * @param {*} promises
   */
  Promise.race = function (promises) {
    return new Promise((resolve, reject) => {
      // 遍历promises，获取每个promise的结果
      promises.forEach((p) => {
        Promise.resolve(p).then(
          (value) => {
            // 第一个异步完成且成功的，就resolve出去
            resolve(value)
          },
          (err) => {
            // 第一个异步完成且失败的，就reject出去
            reject(err)
          }
        )
      })
    })
  }

  // 向外暴露函数
  window.Promise = Promise
})(window)
