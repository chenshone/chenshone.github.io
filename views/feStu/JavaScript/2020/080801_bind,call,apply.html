    手写apply、call、bind

    CHENSHONE 2020/8/8 手写系列JavaScript

    # apply (opens new window)

    /**
 *
 *
 * @param {*} context 调用绑定函数时作为 this 参数传递给目标函数的值。
 * @param {*} args 当目标函数被调用时，被预置入绑定函数的参数列表中的参数。
 * @returns
 */
Function.prototype.myApply = function (context, args) {
  if (typeof this !== 'function') {
    throw TypeError('it is not a function')
  }

  // context如果不传、为 null 或 undefined 时会自动替换为指向全局对象
  context = context || window
  args = args ? args : []

  // 给context新增一个symbol属性用以保存函数，以免覆盖原有属性
  const key = Symbol()
  context[key] = this

  // 通过隐式绑定的方式调用函数
  const result = context[key](...args)

  // 删除添加的属性
  delete context[key]

  return result
}

    # bind (opens new window)

    /**
 *
 *
 * @param {*} context 调用绑定函数时作为 this 参数传递给目标函数的值。
 * @param {*} args1 当目标函数被调用时，被预置入绑定函数的参数列表中的参数。
 */
Function.prototype.myBind = function (context, ...args1) {
  const self = this

  if (typeof this !== 'function') {
    throw TypeError('it is not a function')
  }

  return function newFn(...args2) {
    // 因为返回新的函数,要考虑到使用new去调用,并且new的优先级比较高,所以需要判断是否为new的调用
    // 如果是new调用，那么构造函数的 prototype 属性必然出现在实例对象this的原型链上
    if (this instanceof newFn) {
      return new self(...args1, ...args2)
    }

    // return self.apply(context, [...args1, ...args2])
    context = context || window

    const key = Symbol()
    context[key] = self

    const result = context[key](...args1, ...args2)
    delete context[key]

    return result
  }
}
