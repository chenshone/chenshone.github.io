    CHENSHONE 2020/6/28 golang

    # 定义

    /**
函数名: 由字母、数字和下划线组成，且不能以数字开头。在同一个包内，函数名不能重名。
参数: 由参数名称和参数类型组成，多个参数直接用 , 分隔
返回值: 由返回值名称和返回值类型组成，也可以只写返回值类型，多个返回值必须用 () 包裹，之间用 , 分隔
*/
func 函数名(参数) (返回值) {
  函数体
}

    # defer语句

    defer语句会将其后面跟随的语句进行延迟处理。在defer归属的函数即将返回时，将延迟处理的语句按defer定义的逆序进行执行。

    多用于函数结束之前释放资源(文件句柄、数据库连接、socket连接)

    func main() {
  fmt.Println("start")
  defer fmt.Println(1)
  defer fmt.Println(2)
  defer fmt.Println(3)
  fmt.Println("end")
}

    # 执行时机

    在go中，函数的return语句在底层并不是原子操作，而是分为给返回值赋值和RET两步。而defer语句的执行时机就是在这两者中间。 图片详情 (opens new window)

    # 经典案例

    func f1() int {
	x := 10
	defer func() {
		x++
	}()

	return x
}

func f2() (x int) {
	defer func() {
		x++
	}()

	return 10
}

func f3() (y int) {
	x := 10
	defer func() {
		x++
	}()

	return x
}

func f4() (x int) {
	defer func(x int) {
		x++ // 这里改变的是这个匿名函数内部自己的值，所有没有影响
	}(x)

	return 10
}

func f5() (x int) {
	defer func(x int) int {
    x++ // 这里改变的是这个匿名函数内部自己的值，所有没有影响
    return x
	}(x)

	return 10
}

func f6() (x int) {
	defer func(x *int) {
    (*x)++
	}(&x)

	return 10
}

func main() {
	fmt.Println(f1()) // 10
	fmt.Println(f2()) // 11
  fmt.Println(f3()) // 10
	fmt.Println(f4()) // 10
	fmt.Println(f5()) // 10
	fmt.Println(f6()) // 11
}

    func calc(index string, a, b int) int {
  ret := a + b
  fmt.Println(index, a, b, ret)
  return ret
}

func main() {
  a := 1
  b := 2
  defer calc("1", a, calc("10", a, b))
  a = 0
  defer calc("2", a, calc("20", a, b))
  b = 1
}
