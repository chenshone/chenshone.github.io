    go学习(十六) time包

    CHENSHONE 2020/7/1 golang

    time包提供了时间的显示和测量用的函数。

    # 基本使用

    func main() {
	now := time.Now() // 当前时间
	fmt.Println(now)

	year := now.Year() // 年
	month := now.Month() // 月
	day := now.Day() // 日
	hour := now.Hour() // 小时
	minute := now.Minute() // 分钟
	second := now.Second() // 秒

	fmt.Println(year, month, day, hour, minute, second)
}

    # 时间戳转为时间格式

    使用time.Unix()可以将时间戳转成时间格式

    now := time.Now()
	fmt.Println(now)

	t1 := now.Unix()     // 毫秒时间戳
	t2 := now.UnixNano() // 纳秒时间戳

	fmt.Println(t1, t2)

	timeObj := time.Unix(t1, 0)

	year := timeObj.Year()     // 年
	month := timeObj.Month()   // 月
	day := timeObj.Day()       // 日
	hour := timeObj.Hour()     // 小时
	minute := timeObj.Minute() // 分钟
	second := timeObj.Second() // 秒

	fmt.Println(year, month, day, hour, minute, second)

    const (
	Nanosecond  Duration = 1
	Microsecond          = 1000 * Nanosecond
	Millisecond          = 1000 * Microsecond
	Second               = 1000 * Millisecond
	Minute               = 60 * Second
	Hour                 = 60 * Minute

    # Sub

    求两个时间之间的差值

    func (t Time) Sub(u Time) Duration
    1

    返回一个时间段 t-u, 如果结果超出了Duration可以表示的最大/小值，将返回最大/小值。

    要获取某个时刻往前的时间点，可以使用t.Add(-d)

    # Equal

    func (t Time) Equal(u Time) bool
    1

    该方法和用 t == u 不同，其会考虑时区的影响，因此不同时区标准的时间也可以正确比较。

    # Before

    func (t Time) Before(u Time) bool
    1

    如果 t 的时刻在 u 之前，则返回真，否则返回假

    # After

    func (t Time) After(u Time) bool
    1

    如果 t 的时刻在 u 之后，则返回真，否则返回假

    # 定时器

    使用 time.Tick(时间间隔) 来设置一个定时器，定时器的本质是一个通道(channel)。

    func main() {
  ticker := time.Tick(time.Second) // 定义一个1秒间隔的定时器
  for i := range ticker {
    fmt.Println(i) // 每秒都会执行
  }
}

    # 格式化

    go中有自带的 Format 方法，但是go中的格式化模板不是常见的 YY-MM-DD HH:MM:SS,而是go语言的诞生时间 2006年1月2号15点04分05秒 Mon Jan。

    func main() {
	now := time.Now()

	// 24小时制
	fmt.Println(now.Format("2006-01-02 15:04:05.000 Mon Jan"))

	// 12小时制
	fmt.Println(now.Format("2006-01-02 03:04:05.000 PM Mon Jan"))

	fmt.Println(now.Format("2006/01/02 15:04"))
	fmt.Println(now.Format("03:04 2006/01/02 PM"))
	fmt.Println(now.Format("2006/01/02"))
}
