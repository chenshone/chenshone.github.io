    # 打开和关闭文件

    os.Open()函数能够打开一个文件，返回一个*File和一个err。对得到的文件实例调用close()方法能够关闭文件。

    import (
	"fmt"
	"os"
)

func main() {
	// 读取当前目录下的test.txt文件
	file, err := os.Open("./test.txt")

	if err != nil {
		fmt.Println(err)
		return
	}

	file.Close()
}

    # 读取文件

    file.Read() 函数能够读取一个文件

    func (*os.File).Read(b []byte) (n int, err error)
    1

    接收一个字节切片，返回读取的字节数和可能出现的错误，读到文件末尾时会返回 0 和 io.EOF。

    package main

import (
	"fmt"
	"io"
	"os"
)

func main() {
	// 读取当前目录下的test.txt文件
	file, err := os.Open("./test.txt")

	if err != nil {
		fmt.Println(err)
		return
	}

	// 使用defer 程序结束时关闭文件
	defer file.Close()

	// 读取文件

	var tmp [128]byte
	// 循环读取
	for {
		n, err := file.Read(tmp[:])

		if err == io.EOF {
			fmt.Println("读完了")
			return
		}

		if err != nil {
			fmt.Println(err)
			return
		}

		fmt.Printf("读了%d个字节", n)
		fmt.Println(string(tmp[:n]))

		// 如果小于128，说明本次是读取的最后一次
		if n < 128 {
			return
		}
	}
}

    # bufio读取文件

    bufio是在file的基础上封装了一层API，支持更多的功能。

    package main

import (
	"bufio"
	"fmt"
	"io"
	"os"
)

func main() {
	// 读取当前目录下的test.txt文件
	file, err := os.Open("./test.txt")

	if err != nil {
		fmt.Println(err)
		return
	}

	// 使用defer 程序结束时关闭文件
	defer file.Close()

	// 读取文件
	reader := bufio.NewReader(file)
	// 循环读取
	for {
		// 读到换行结束一次的读取(每次读取一行)
		line, err := reader.ReadString('\n') // 注意：是字符要用单引号

		if err == io.EOF {
			fmt.Println("读完了")
			return
		}

		if err != nil {
			fmt.Println(err)
			return
		}

		fmt.Print(line)
	}
}

    # ioutil读取整个文件

import (
	"fmt"
	"io/ioutil"
)

func main() {
	content, err := ioutil.ReadFile("./test.txt")

	if err != nil {
		fmt.Println(err)
		return
	}

	fmt.Println(string(content))
}

    # 文件写入

    os.OpenFile()函数能够以指定模式打开文件，从而实现文件写入相关功能。

    func os.OpenFile(name string, flag int, perm os.FileMode) (*os.File, error)
    1

    name: 要打开的文件名

    flag: 打开文件的模式

    模式 含义
    os.O_RDONLY 只读
    os.O_RDWR 读写
    os.O_WRONLY 只写
    os.O_CREATE 创建文件
    os.O_TRUNC 清空
    os.O_APPEND 追加

    这些常量其实是一些数字

    perm: 文件权限，一个八进制数。r(读) 04、 w(写) 02、 x(执行) 01。

    # Write和WriteString

    func main() {
  // flag可以用或运算实现多个模式
	file, err := os.OpenFile("test.txt", os.O_CREATE|os.O_TRUNC|os.O_WRONLY, 0666)
	if err != nil {
		fmt.Println(err)
		return
	}

	defer file.Close()

	str := "hello world"

	file.Write([]byte(str)) // 写入字节切片
	file.WriteString(str)   // 直接写入字符串
}

    # bufio.NewWriter

    func main() {
	file, err := os.OpenFile("test.txt", os.O_CREATE|os.O_TRUNC|os.O_WRONLY, 0666)
	if err != nil {
		fmt.Println(err)
		return
	}

	defer file.Close()

	writer := bufio.NewWriter(file)
	str := "hello world"

	for i := 0; i < 10; i++ {
		// 将数据写入缓存
		writer.WriteString(str)
	}
	// 将缓存内容写入文件
	writer.Flush()
}
