    go学习(十二) 结构体

    CHENSHONE 2020/6/29 golang

    基本数据类型只能保存同类型的数据，对于一组包含不同类型的数据，可以使用结构体来保存。

    结构体也是一种类型。go中通过结构体来实现面向对象。

    # 类型别名和自定义类型

    使用type关键字可以来定义自定义类型和类型别名

    // myInt是自定义类型，具有int的特性
type myInt int
// myInt1是int的类型别名
type myInt1 = int

func main() {
  var a myInt
  var b myInt1

  fmt.Printf("%T", a) // main.myInt
  fmt.Printf("%T", b) // int
}

    # 实例化指针类型结构体

    # 使用new关键字

    可以通过new关键字来实例化结构体，得到的就是结构体的地址

    type person struct {
	name string
	age  int
}

func main() {
	var a = new(person)
  fmt.Printf("%T\n", a) // *main.person
  fmt.Printf("%#v\n", a) // &main.person{name:"", age:0}

  a.name = "chenshone"
	a.age = 20

	fmt.Printf("%#v\n", a) // &main.person{name:"chenshone", age:20}
}

    # 结构体的内存地址

    结构体是一块连续的内存地址

    type foo struct {
	a int8
	b int8
	c int8
}

func main() {
	var bar = foo{
		a: int8(1),
		b: int8(2),
		c: int8(3),
	}

	fmt.Printf("%p\n", &(bar.a)) // 0xc0000a0068
	fmt.Printf("%p\n", &(bar.b)) // 0xc0000a0069
	fmt.Printf("%p\n", &(bar.c)) // 0xc0000a006a
}

    # 构造函数

    go中结构体没有构造函数，但可以模拟

    type person struct {
	name string
	age  int
}

// 当结构体比较大的时候尽量使用结构体指针，减少程序的内存开销
func newPerson(name string, age int) *person {
	return &person{
		name: name,
		age:  age,
	}
}

func main() {
	p1 := newPerson("chenshone", 20)
	p2 := newPerson("foo", 18)

	fmt.Println(p1) // &{chenshone 20}
	fmt.Println(p2) // &{foo 18}
}

    # 方法

    方法是作用于特定类型的函数

    只能给自己包里的类型添加方法，不能给别的包里的类型添加方法。

    如果非要给别的包的类型添加方法，可以使用自定义类型来添加方法

    func (接收者变量 接收者类型) 方法名(参数列表) (返回参数) {
  函数体
}
    1
    2
    3

    接收者就是类似于其他语言的this、self，表示这个方法是作用于谁的。

    接收者类型可以是值类型也可以是指针类型，一般都是使用指针。

    func (p *person) speak() {
	fmt.Printf("%v在说话", p.name)
}

func main() {
	p1 := newPerson("chenshone", 20)
	p1.speak() // chenshone在说话
}

    # 匿名字段

    结构体的成员在声明时，如果没有字段名只有类型，这就是匿名字段。

    匿名字段默认采用类型名作为字段名，因为结构体要求字段名必须唯一，所以一个结构体中同种类型的匿名字段只能有一个。

    type person struct {
	string
	int
}

func main() {
	p1 := person{
		"chenshone",
		20,
	}

	fmt.Println(p1.int)    // 20
	fmt.Println(p1.string) // chenshone
}

    # 结构体嵌套

    type address struct {
	city string
}

type person struct {
	name string
	age  int
	addr address
}

func main() {
	p1 := person{
		name: "chenshone",
		age:  20,
		addr: address{
			city: "China",
		},
	}

	fmt.Println(p1.name)  // 20
	fmt.Println(p1.age)   // 20
	fmt.Println(p1.addr.city) // China
}

    # 结构体的 "继承"

    结构体模拟其他语言中类的继承

    type animal struct {
	name string
}

func (a animal) move() {
	fmt.Printf("%s会动\n", a.name)
}

type dog struct {
	feet uint8
	animal
}

func (d dog) bark() {
	fmt.Printf("%s在叫", d.name)
}

func main() {
	d1 := dog{
		feet: 50,
		animal: animal{
			name: "foo",
		},
	}

	d1.move() // foo会动
	d1.bark() // foo在叫
}

    # 结构体与JSON

    JSON是前后端交互的重要格式。在go中可以实现结构体和JSON之间的序列化与反序列化的转换

    type person struct {
  // 向外暴露这两个成员变量
	Name string `json:"name" db:"name"`
	Age  int    `json:"age"`
}

func main() {
	p1 := person{
		Name: "chenshone",
		Age:  20,
	}

  // 序列化
	b, err := json.Marshal(p1)

	if err != nil {
		fmt.Println(err)
		return
	}
  fmt.Printf("%#v\n",string(b)) // "{\"name\":\"chenshone\",\"age\":20}"

  // 反序列化
  str := `{"name":"chenshone","age":20}`
	var p2 person
	json.Unmarshal([]byte(str), &p2)

	fmt.Printf("%#v\n",p2) // main.person{Name:"chenshone", Age:20}
}
