    go学习(十三) 接口

    CHENSHONE 2020/6/30 golang

    接口定义了一个对象的行为规范，只定义规范不实现，由具体的对象来实现规范的细节。

    接口是一种类型。

    # 定义

    type 接口名 interface {
  方法名1(参数1, 参数2...) (返回值1, 返回值2...)
  方法名2(参数1, 参数2...) (返回值1, 返回值2...)
  ...
}

    # 实现

    一个变量如果实现了接口中规定的所有方法，那么就可以说这个变量实现了这个接口，可以成为这个接口类型的变量。

    type animal interface {
	move()
	eat()
}

type dog struct {
	name string
}

func (d dog) move() {
	fmt.Println("dog moving")
}

func (d dog) eat() {
	fmt.Println("dog eating")
}

type cat struct {
	name string
}

func (c cat) move() {
	fmt.Println("cat moving")
}

func (c cat) eat() {
	fmt.Println("cat eating")
}

func main() {
	d1 := dog{
		name: "foo",
	}

	c1 := cat{
		name: "bar",
	}

  // 定义接口类型的变量
	var a1, a2 animal

	a1 = d1
	a2 = c1

	fmt.Println(a1)
  fmt.Println(a2)

  fmt.Printf("%T\n", a1) // main.dog
	fmt.Printf("%T\n", a2) // main.cat
}

    上述a1、a2打印的类型还是他们原来的类型(dog、cat)。因为接口是引用类型，在底层的实现分为动态类型和动态值两部分，动态类型保存的是赋值的变量原本的类型，动态值保存的就是赋值过来的值。这样就实现了接口变量能存储不同的值。

    # 指针接收者与值接收者实现接口的区别

    值接收者实现接口，结构体类型和结构体指针类型的变量都能存

    指针接收者实现接口，只能存结构体指针类型的变量

    type animal interface {
	move()
	eat()
}

type dog struct {
	name string
}

func (d dog) move() {
	fmt.Println("dog moving")
}

func (d dog) eat() {
	fmt.Println("dog eating")
}

func main() {
	d1 := dog{
		name: "foo",
	}

	d2 := &dog{
		name: "bar",
	}

	var a1, a2 animal

	a1 = d1
	a2 = d2

	fmt.Println(a1)
	fmt.Println(a2) // &{bar}

	fmt.Printf("%T\n", a1)
	fmt.Printf("%T\n", a2) // *main.dog
}

    # 空接口

    空接口是指没有定义任何方法的接口, 因此任何类型都实现了空接口。

    空接口类型的变量可以存储任意类型的变量。

    # 定义

    // 一般来说空接口直接匿名即可
var x interface{}
    1
    2

    # 应用

    # 作为函数参数

    // x参数可以接收任意类型
func foo(x interface{}) {
  fmt.Printf("%T", x)
}

foo["a1"] = 233

foo["a3"] = true

    # 类型断言

    空接口可以存储任意类型的值，那么如何获取具体的数据呢？可以使用类型断言

    
/**
x: 表示类型为interface{}的变量
T: 表示断言x可能是的类型
*/
x.(T)

    # 示例

    func foo(x interface{}) {
	_, ok := x.(string)

	if ok {
		fmt.Println("x是string类型")
	} else {
		fmt.Println("x不是string类型")
	}
}

func main() {
	foo(233) // x不是string类型
}

    func foo(x interface{}) {
	switch v := x.(type) {
	case string:
		fmt.Printf("x是string类型，value: %v\n", v)
	case int:
		fmt.Printf("x是int类型，value: %v\n", v)
	case bool:
		fmt.Printf("x是string类型，value: %v\n", v)
	default:
		fmt.Printf("x是其他类型，value: %v\n", v)
	}
}

func main() {
	foo(233) // x是int类型，value: 233
	foo("233") // x是string类型，value: 233
	foo(true) // x是bool类型，value: true
}
